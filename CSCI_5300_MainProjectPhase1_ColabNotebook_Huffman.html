#
# At this point, we need to convert all the string values (such as Make, Brand, Transmission)
# to number values.  I've created a quick function to do this.  All that needs to be done
# is that every column to be converted needs a "Series" to be defined based on the unique values
# in that table's columns.  I should be able to reverse lookup this Series just as easily.
#
# There is probably a better way to do this, but this works reliably
#
# Make
makeSeries = pd.Series(df.make.unique())
def MakeConvertColToIndexedNumbers(Val):
  return makeSeries[makeSeries==Val].first_valid_index()
df['make'] = df['make'].apply(MakeConvertColToIndexedNumbers)

# Model
modelSeries = pd.Series(df.model.unique())
def ModelConvertColToIndexedNumbers(Val):
  return modelSeries[modelSeries==Val].first_valid_index()
df['model'] = df['model'].apply(ModelConvertColToIndexedNumbers)

# Body
bodySeries = pd.Series(df.body.unique())
def bodyConvertColToIndexedNumbers(Val):
  return bodySeries[bodySeries==Val].first_valid_index()
df['body'] = df['body'].apply(bodyConvertColToIndexedNumbers)

# Trans
transmissionSeries = pd.Series(df.transmission.unique())
def transmissionConvertColToIndexedNumbers(Val):
  return transmissionSeries[transmissionSeries==Val].first_valid_index()
df['transmission'] = df['transmission'].apply(transmissionConvertColToIndexedNumbers)

# State
stateSeries = pd.Series(df.state.unique())
def stateConvertColToIndexedNumbers(Val):
  return stateSeries[stateSeries==Val].first_valid_index()
df['state'] = df['state'].apply(stateConvertColToIndexedNumbers)

# Color
colorSeries = pd.Series(df.color.unique())
def colorConvertColToIndexedNumbers(Val):
  return colorSeries[colorSeries==Val].first_valid_index()
df['color'] = df['color'].apply(colorConvertColToIndexedNumbers)

# Interior
interiorSeries = pd.Series(df.interior.unique())
def interiorConvertColToIndexedNumbers(Val):
  return interiorSeries[interiorSeries==Val].first_valid_index()
df['interior'] = df['interior'].apply(interiorConvertColToIndexedNumbers)
